13

18       @Override
19       public void start(Stage primaryStage) {
20           Pane pane = new Pane();
21           Polygon polygon = new Polygon();
22           pane.getChildren().add(polygon);
23   
24           polygon.setFill(Color.GOLD);
25           polygon.setStroke(Color.BLACK);
26   
27           ObservableList<Double> obsPoly = polygon.getPoints();
28   
29           final double WIDTH = 200, HEIGHT = 200;
30   
31           double centerX = WIDTH / 2, centerY = HEIGHT / 2;
32           double radius = Math.min(WIDTH, HEIGHT) * .4;
33   
34           for(int i = 0; i < 6; i++) {
35               obsPoly.add(centerX + radius * Math.cos(2* i * Math.PI /6)) ;
36               obsPoly.add(centerY - radius * Math.sin(2* i * Math.PI /6)) ;
37           }
38           primaryStage.setTitle("Polygon");
39           primaryStage.setScene(new Scene(pane, WIDTH, HEIGHT));
40           primaryStage.show();
41       }
